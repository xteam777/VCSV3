SET NOCOUNT ON;

SET TRANSACTION ISOLATION LEVEL READ COMMITTED;

BEGIN TRANSACTION;

DELETE AC
FROM Accounts_Connections AS AC
	INNER JOIN Accounts AS A ON AC.AccountUID = A.UID AND A.IsCommerce = 1 AND AC.Finalized = 1;

DELETE DC
FROM Devices_Connections AS DC
	INNER JOIN Devices AS D ON DC.DeviceUID = D.UID AND D.IsCommerce = 1 AND DC.Finalized = 1;

COMMIT TRANSACTION;
